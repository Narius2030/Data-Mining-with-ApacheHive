CREATE TABLE Fact_Inventory_Analysis (
                                        inventory_key INT,
                                        rental_id INT,
                                        orderdate_key STRING,
                                        remaining INT,
                                        Total_Rental_Amount FLOAT
                                    )
                                    ROW FORMAT DELIMITED
                                    FIELDS TERMINATED BY ','
